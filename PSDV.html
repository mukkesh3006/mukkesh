def standard_deviation(data):

#Character frequency
import matplotlib.pyplot as plt
def character_frequency(input_string):
    input_string = input_string.lower()
    frequency_dict = {}
    for char in input_string:
        if char.isalpha():
            if char in frequency_dict:
                frequency_dict[char] += 1
            else:
                frequency_dict[char] = 1

    return frequency_dict
input_string=input()
freq_dict = character_frequency(input_string)
print("Character frequencies:", freq_dict)

char=[]
for characters in freq_dict.keys():
  char=char+[characters]

freq=[]
for frequencies in freq_dict.values():
  freq=freq+[frequencies]
plt.bar(char, freq, color='green')
plt.title("Character Frequency")
plt.xlabel("Character")
plt.ylabel("Frequency")
plt.show()

#string operation
import matplotlib.pyplot as plt

def string_operations(strings):
    longest_string = max(strings, key=len) if strings else ""
    vowels = {'a', 'e', 'i', 'o', 'u'}
    strings_starting_with_vowel = sum(1 for s in strings if s and s[0].lower() in vowels)
    string_lengths = [len(s) for s in strings]
    plt.hist(string_lengths, bins=range(1, max(string_lengths) + 2), color='blue', edgecolor='black')
    plt.title("Distribution of String Lengths")
    plt.xlabel("String Length")
    plt.ylabel("Frequency")
    plt.show()
    return longest_string, strings_starting_with_vowel
input_strings = input("Enter a comma-separated list of strings: ").split(',')
input_strings = [s.strip() for s in input_strings]
longest, vowel_count = string_operations(input_strings)
print("Longest string:", longest)
print("Number of strings starting with a vowel:", vowel_count)

#sorting dictionaries
def sort_dicts_by_key(data, key):
    try:
        sorted_data = sorted(data, key=lambda x: x[key])
        return sorted_data
    except KeyError:
        print(f"Error: Key '{key}' not found in some dictionaries.")
        return data
    except TypeError:
        print("Error: Input data is not a list of dictionaries.")
        return data
data = [{'name': 'Alice', 'age': 25}, {'name': 'Bob', 'age': 22}, {'name': 'Charlie', 'age': 30}]
key_to_sort_by = 'age'

def filter_and_sort_words(input_string):
    words = input_string.split()

    plt.xlabel("Words")
    plt.ylabel("Length")

#Anagram finder
def find_anagrams(word, words):
    sorted_word = sorted(word)
    anagrams = [w for w in words if sorted(w) == sorted_word]
    return anagrams

#Set Operation
from matplotlib_venn import venn2
import matplotlib.pyplot as plt
def set_operations(list1, list2):
    set1 = set(list1)
    set2 = set(list2)
    intersection = set1 & set2
    union = set1 | set2
    symmetric_difference = set1 ^ set2
    venn = venn2([set1, set2], ('Set 1', 'Set 2'))
    plt.title("Venn Diagram of Set Operations")
    plt.show()
    return intersection, union, symmetric_difference
list1 = [1, 2, 3, 4, 5]
list2 = [4, 5, 6, 7, 8]
intersection, union, symmetric_difference = set_operations(list1, list2)
print("Intersection:", intersection)
print("Union:", union)
print("Symmetric Difference:", symmetric_difference)

#File Operation
def create_and_write_file(filename, numbers):
    with open(filename, 'w') as file:
        for number in numbers:
            file.write(f"{number}\n")

def read_and_sum_file(filename):
    total = 0
    with open(filename, 'r') as file:
        for line in file:
            total += int(line.strip())
    return total
filename = "numbers.txt"
numbers = [10, 20, 30, 40, 50]
create_and_write_file(filename, numbers)
print(f"Numbers written to {filename}.")
total_sum = read_and_sum_file(filename)
print(f"The sum of the numbers in {filename} is: {total_sum}")
